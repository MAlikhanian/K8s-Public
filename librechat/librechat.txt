# 1. Create the required secret with all necessary keys
kubectl create secret generic librechat-credentials-env \
  --from-literal=CREDS_KEY=$(openssl rand -hex 16) \
  --from-literal=CREDS_IV=$(openssl rand -hex 8) \
  --from-literal=JWT_SECRET=$(openssl rand -hex 16) \
  --from-literal=JWT_REFRESH_SECRET=$(openssl rand -hex 16) \
  --from-literal=MEILI_MASTER_KEY=$(openssl rand -hex 16) \
  --from-literal=OPENAI_API_KEY="sk-your-actual-openai-key-here" \
  --namespace librechat

# 2. Verify the secret was created
kubectl get secret -n librechat librechat-credentials-env
kubectl describe secret -n librechat librechat-credentials-env
